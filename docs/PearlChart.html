PearlChart

A pearl chart component.

Constructor

new PearlChart(options)

Source:
  • PearlChart.js, line 67
Example
var callbackFunc = function(data) {
  console.log(data);
}

var PearlChartExample = new OECDCharts.PearlChart({
  container: '#PearlChartExample',
  extent: [300, 600],
  title: 'Pearl Chart',
  renderInfoButton: true,
  showTicks: true,
  showLabels: false,
  colorLabels: true,
  callback: callbackFunc,
  data: [
    {
      value: 410,
      color: '#900c3f'
    },
    {
      value: 520,
      color: '#189aa8'
    }
  ],
  labelFormat: function(val) {
    return Math.round(val) + '$';
  }
});
